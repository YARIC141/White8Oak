// Тёмная тема
const themeToggle = document.getElementById('themeToggle');
if (themeToggle) {
    if (localStorage.getItem('theme') === 'dark') {
        document.body.classList.add('dark');
        themeToggle.innerHTML = '☀️ Светлая';
    }
    
    themeToggle.addEventListener('click', () => {
        document.body.classList.toggle('dark');
        const isDark = document.body.classList.contains('dark');
        themeToggle.innerHTML = isDark ? '☀️ Светлая' : '🌙 Тёмная';
        localStorage.setItem('theme', isDark ? 'dark' : 'light');
    });
}

// Функция для определения базового пути на GitHub Pages
function getBasePath() {
    if (window.location.hostname === 'yaric141.github.io') {
        return '/White8Oak';
    }
    return '';
}

// Загрузка товаров из data/products.json
async function loadProducts() {
    try {
        const basePath = getBasePath();
        const url = `${basePath}/data/products.json?_=${Date.now()}`;
        console.log('🔄 Загрузка товаров из:', url);
        
        const response = await fetch(url);
        
        if (response.ok) {
            const data = await response.json();
            console.log('✅ Загружено товаров:', data.products?.length || 0);
            return data;
        } else {
            console.error('❌ Ошибка загрузки, статус:', response.status);
            return { products: [], categories: [] };
        }
    } catch (error) {
        console.error('❌ Ошибка:', error);
        return { products: [], categories: [] };
    }
}

// Загрузка и отображение каталога
async function loadCatalog() {
    console.log('🔄 Загрузка каталога...');
    
    const container = document.getElementById('catalog-products');
    if (!container) return;
    
    container.innerHTML = '<div style="text-align: center; padding: 60px;">🔄 Загрузка товаров...</div>';
    
    const data = await loadProducts();
    
    if (!data.products || data.products.length === 0) {
        container.innerHTML = `
            <div style="text-align: center; padding: 60px;">
                <i class="fas fa-box-open" style="font-size: 48px; opacity: 0.5;"></i>
                <h3>📦 Нет товаров</h3>
                <p>Товары появятся здесь после добавления через админ-панель</p>
            </div>
        `;
        return;
    }
    
    const urlParams = new URLSearchParams(window.location.search);
    const activeCategory = urlParams.get('category');
    
    const filterContainer = document.getElementById('categories-filter');
    if (filterContainer && data.categories) {
        filterContainer.innerHTML = `
            <button class="filter-btn ${!activeCategory ? 'active' : ''}" data-category="all">
                Все (${data.products.length})
            </button>
            ${data.categories.map(cat => {
                const count = data.products.filter(p => p.category === cat.id).length;
                return `<button class="filter-btn ${activeCategory === cat.id ? 'active' : ''}" data-category="${cat.id}">
                    <i class="fas ${cat.icon}"></i> ${cat.name} (${count})
                </button>`;
            }).join('')}
        `;
        
        document.querySelectorAll('.filter-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                const category = btn.dataset.category;
                const basePath = getBasePath();
                if (category === 'all') {
                    window.location.href = `${basePath}/catalog.html`;
                } else {
                    window.location.href = `${basePath}/catalog.html?category=${category}`;
                }
            });
        });
    }
    
    let products = data.products;
    if (activeCategory && activeCategory !== 'all') {
        products = data.products.filter(p => p.category === activeCategory);
    }
    
    if (products.length === 0) {
        container.innerHTML = `<div style="text-align:center; padding:60px;">Нет товаров в этой категории</div>`;
    } else {
        const basePath = getBasePath();
        container.innerHTML = products.map(product => {
            const hasImage = product.image && product.image !== '';
            const icon = getCategoryIcon(product.category);
            const imagePath = hasImage ? `${basePath}/images/products/${product.image}` : '';
            
            return `
                <div class="product-card" onclick="location.href='${basePath}/product.html?id=${product.id}'">
                    <div class="product-img">
                        ${hasImage ? 
                            `<img src="${imagePath}" alt="${product.title}" 
                                  style="width:100%; height:100%; object-fit:cover;"
                                  onerror="this.parentElement.innerHTML='<i class=\\'fas ${icon}\\' style=\\'font-size:64px;\\'></i>'">` :
                            `<i class="fas ${icon}" style="font-size:64px;"></i>`
                        }
                    </div>
                    <div class="product-info">
                        <div class="product-title">${escapeHtml(product.title)}</div>
                        <div class="product-price">${product.price.toLocaleString()} ₽</div>
                        ${!product.inStock ? '<div class="out-of-stock">Нет в наличии</div>' : ''}
                    </div>
                </div>
            `;
        }).join('');
    }
}

function getCategoryIcon(categoryId) {
    const icons = {
        'tables': 'fa-table', 'chairs': 'fa-chair', 'beds': 'fa-bed',
        'shelves': 'fa-layer-group', 'nightstands': 'fa-draw-polygon', 'chests': 'fa-archive'
    };
    return icons[categoryId] || 'fa-couch';
}

function getCategoryName(categoryId) {
    const categories = {
        'tables': 'Столы', 'chairs': 'Стулья', 'beds': 'Кровати',
        'shelves': 'Стеллажи', 'nightstands': 'Тумбы', 'chests': 'Комоды'
    };
    return categories[categoryId] || 'Мебель';
}

function escapeHtml(str) {
    if (!str) return '';
    return str.replace(/[&<>]/g, function(m) {
        if (m === '&') return '&amp;';
        if (m === '<') return '&lt;';
        if (m === '>') return '&gt;';
        return m;
    });
}

// Генерация ID заказа
function generateOrderId() {
    return 'ORD-' + Date.now() + '-' + Math.random().toString(36).substr(2, 9).toUpperCase();
}

async function sendOrderEmail(orderData) {
